import 'dart:io';

import 'package:cloud_firestore/cloud_firestore.dart';
import 'package:firebase_auth/firebase_auth.dart';
import 'package:flutter/material.dart';
import 'package:image_picker/image_picker.dart';
import 'package:intl/intl.dart';
import 'package:firebase_storage/firebase_storage.dart';

class PayDuesPage extends StatefulWidget {
  const PayDuesPage({super.key});

  @override
  State<PayDuesPage> createState() => _PayDuesPageState();
}

class _PayDuesPageState extends State<PayDuesPage> {
  final FirebaseFirestore _db = FirebaseFirestore.instance;
  final FirebaseStorage _storage = FirebaseStorage.instance;
  final String? _userId = FirebaseAuth.instance.currentUser?.uid;

  String? _selectedPaymentId;
  File? _proofFile;
  bool _loading = false;

  final ImagePicker _picker = ImagePicker();

  Future<void> _pickProof() async {
    final XFile? pickedFile = await _picker.pickImage(source: ImageSource.gallery);
    if (pickedFile != null) {
      setState(() {
        _proofFile = File(pickedFile.path);
      });
    }
  }

  Future<void> _submitPayment() async {
    if (_selectedPaymentId == null || _proofFile == null) {
      ScaffoldMessenger.of(context).showSnackBar(
        const SnackBar(content: Text('Please select a due and upload proof.')),
      );
      return;
    }

    setState(() => _loading = true);

    try {
      // 1️⃣ Upload proof to Firebase Storage
      final storageRef = _storage.ref().child('payment_proofs/${_selectedPaymentId}_${DateTime.now().millisecondsSinceEpoch}');
      final uploadTask = await storageRef.putFile(_proofFile!);
      final proofUrl = await uploadTask.ref.getDownloadURL();

      // 2️⃣ Update Firestore payment document
      await _db.collection('payments').doc(_selectedPaymentId).update({
        'proofRef': proofUrl,
        'status': 'Pending Review',
        'datePaid': FieldValue.serverTimestamp(),
      });

      ScaffoldMessenger.of(context).showSnackBar(
        const SnackBar(content: Text('Payment submitted for review!')),
      );

      // Reset selections
      setState(() {
        _selectedPaymentId = null;
        _proofFile = null;
      });
    } catch (e) {
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(content: Text('Error submitting payment: $e')),
      );
    } finally {
      setState(() => _loading = false);
    }
  }

  @override
  Widget build(BuildContext context) {
    if (_userId == null) {
      return const Scaffold(
        body: Center(child: Text('User not logged in.')),
      );
    }

    return Scaffold(
      appBar: AppBar(
        title: const Text('Pay Dues'),
        backgroundColor: Colors.blueAccent,
      ),
      body: StreamBuilder<QuerySnapshot>(
        stream: _db
            .collection('payments')
            .where('userId', isEqualTo: _userId)
            .where('status', isEqualTo: 'Due') // only pending dues
            .orderBy('dateDue')
            .snapshots(),
        builder: (context, snapshot) {
          if (snapshot.connectionState == ConnectionState.waiting) {
            return const Center(child: CircularProgressIndicator());
          }

          if (!snapshot.hasData || snapshot.data!.docs.isEmpty) {
            return const Center(
              child: Text('No pending dues to pay.', style: TextStyle(color: Colors.black54)),
            );
          }

          final dues = snapshot.data!.docs;

          return Padding(
            padding: const EdgeInsets.all(12),
            child: Column(
              children: [
                Expanded(
                  child: ListView.separated(
                    itemCount: dues.length,
                    separatorBuilder: (_, __) => const SizedBox(height: 8),
                    itemBuilder: (context, index) {
                      final data = dues[index].data() as Map<String, dynamic>;
                      final paymentId = data['paymentId'];
                      final categoryName = data['categoryName'] ?? 'Unnamed';
                      final amount = data['amountOwed'] ?? 0;
                      final dueDate = (data['dateDue'] as Timestamp).toDate();
                      final billingPeriod = data['billingPeriod'] ?? '';

                      final selected = _selectedPaymentId == paymentId;

                      return Card(
                        elevation: 1,
                        shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(8)),
                        child: ListTile(
                          contentPadding: const EdgeInsets.symmetric(horizontal: 12, vertical: 8),
                          leading: Radio<String>(
                            value: paymentId,
                            groupValue: _selectedPaymentId,
                            onChanged: (value) {
                              setState(() => _selectedPaymentId = value);
                            },
                          ),
                          title: Text(categoryName, style: const TextStyle(fontWeight: FontWeight.bold)),
                          subtitle: Text(
                            'Amount: ₱${NumberFormat('#,##0.00').format(amount)} • Due: ${DateFormat('MMM d, yyyy').format(dueDate)} • $billingPeriod',
                            style: const TextStyle(fontSize: 12, color: Colors.black54),
                          ),
                        ),
                      );
                    },
                  ),
                ),

                const SizedBox(height: 12),
                _proofFile != null
                    ? Text('Selected proof: ${_proofFile!.path.split('/').last}')
                    : const Text('No proof selected.', style: TextStyle(color: Colors.black54)),

                const SizedBox(height: 8),
                Row(
                  children: [
                    Expanded(
                      child: ElevatedButton.icon(
                        onPressed: _pickProof,
                        icon: const Icon(Icons.upload_file),
                        label: const Text('Upload Proof'),
                        style: ElevatedButton.styleFrom(backgroundColor: Colors.blueAccent),
                      ),
                    ),
                  ],
                ),
                const SizedBox(height: 12),
                Row(
                  children: [
                    Expanded(
                      child: ElevatedButton(
                        onPressed: _loading ? null : _submitPayment,
                        child: _loading
                            ? const SizedBox(height: 20, width: 20, child: CircularProgressIndicator(color: Colors.white, strokeWidth: 2))
                            : const Text('Submit Payment'),
                        style: ElevatedButton.styleFrom(backgroundColor: Colors.green),
                      ),
                    ),
                  ],
                ),
                const SizedBox(height: 12),
              ],
            ),
          );
        },
      ),
    );
  }
}
